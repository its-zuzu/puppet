# CTF Flag Submission Rate Limits
# ---------------------------------
# Optimized for ~400 concurrent users
# Strategy: Per-User/Challenge "Guess Protection" + Global DoS Protection

# --- 1. Per-User Anti-Brute-Force ---
# How many guesses allowed per challenge
FLAG_SUBMIT_MAX_ATTEMPTS=5

# The rolling window for those attempts (in seconds)
# User can make 5 attempts every 60 seconds
FLAG_SUBMIT_WINDOW=60

# Penalty time if limit is hit (in seconds)
# User is BLOCKED from this specific challenge for 30s
FLAG_SUBMIT_COOLDOWN=30

# --- 2. Global DoS Protection ---
# Limits TOTAL submission requests per IP address across all challenges
# Prevents a script from spamming the server
FLAG_SUBMIT_RATE_WINDOW=1m
FLAG_SUBMIT_RATE_MAX=50

# Note:
# - These limits are backed by Redis.
# - If Redis fails, the system "fails open" (allows submissions) but logs errors.
# - Keep these loose enough for manual play, tight enough to stop scripts.
